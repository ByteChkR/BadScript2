103 {
104 BadPropertyVisibility vis = BadScope.GetPropertyVisibility(propName);
105
106 if (caller != null)
107 {
108 if (caller.ClassObject == null)
109 {
110 visibility = BadPropertyVisibility.Public;
111 }
112 else if (caller.ClassObject == this)
113 {
114 visibility = BadPropertyVisibility.All;
115 }
116 else if (caller.ClassObject.InheritsFrom(Prototype))
117 {
118 visibility = BadPropertyVisibility.Public | BadPropertyVisibility.Protected;
119 }
120 else
121 {
122 visibility = BadPropertyVisibility.Public;
123 }
124 }
125
126 if (!HasProperty(propName))
127 {
128 throw BadRuntimeException.Create(caller,
129 $"Property {propName} not found in class {Name} or any of its base classes");
130 }
131
132 if ((vis & visibility) == 0)
133 {
134 throw BadRuntimeException.Create(caller,
135 $"Property {Name}.{propName} is not visible from {caller?.Name ?? "global"}");
136 }
137
138 if (Scope.GetTable().InnerTable.ContainsKey(propName))
139 {
140 return BadObjectReference.Make($"{Name}.{propName}",
141 () => Scope.GetTable().InnerTable[propName],
142 (o, _) =>
143 {
144 if (Scope.GetTable().InnerTable.ContainsKey(propName))
145 {
146 BadPropertyInfo info = Scope.GetTable().GetPropertyInfo(propName);
147
148 if (Scope.GetTable().InnerTable[propName] != Null && info.IsReadOnly)
149 {
150 throw BadRuntimeException.Create(caller, $"{Name}.{propName} is read-only");
151 }
152
153 if (info.Type != null && !info.Type.IsAssignableFrom(o))
154 {
155 throw BadRuntimeException.Create(caller,
156 $"Cannot assign object {o.GetType().Name} to property '{propName}' of type '{info.Type.Name}'");
157 }
158 }
159
160 Scope.GetTable().InnerTable[propName] = o;
161 });
162 }

167 visibility & ~BadPropertyVisibility.Private & BadPropertyVisibility.All,
168 caller); //Allow public, protected
169 }
