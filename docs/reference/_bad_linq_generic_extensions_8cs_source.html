13 private static TOut InnerSelect<T, TOut>(string varName, BadExpression query, T o)
14 {
15 BadExecutionContext ctx = BadLinqCommon.PredicateContextOptions.Build();
16
17 if (o is BadObject bo)
18 {
19 ctx.Scope.DefineVariable(varName, bo);
20 }
21 else
22 {
23 if (BadObject.CanWrap(o))
24 {
25 ctx.Scope.DefineVariable(varName, BadObject.Wrap(o));
26 }
27 else
28 {
29 ctx.Scope.DefineVariable(varName, new BadReflectedObject(o!));
30 }
31 }
32
33 BadObject r = BadObject.Null;
34
35 foreach (BadObject o1 in query.Execute(ctx))
36 {
37 if (ctx.Scope.IsError)
38 {
39 throw new Exception($"Error in LINQ Where: {varName} => {query} : {ctx.Scope.Error!.ToSafeString()}");
40 }
41
42 r = o1;
43 }
44
45 r = r.Dereference();
46
47 if (r is TOut t)
48 {
49 return t;
50 }
51
52 if (r.CanUnwrap())
53 {
54 return r.Unwrap<TOut>();
55 }
56
57 if (r is BadReflectedObject ro && ro.Type.IsAssignableFrom(typeof(TOut)))
58 {
59 return (TOut)ro.Instance;
60 }
61
62 throw new Exception(
63 $"Error in LINQ Where: {varName} => {query} : {r.ToSafeString()} is not a {typeof(TOut).Name}");
64 }
