73 private void LoadApis(BadExecutionContext context)
74 {
75 foreach (BadInteropApi api in m_Apis)
76 {
77 BadTable target;
78 if (context.Scope.HasLocal(api.Name) && context.Scope.GetVariable(api.Name).Dereference() is BadTable table)
79 {
80 target = table;
81 }
82 else
83 {
84 target = new BadTable();
85 context.Scope.DefineVariable(api.Name, target);
86 }
87
88 api.Load(target);
89 }
90
91 foreach (BadClassPrototype type in BadNativeClassBuilder.NativeTypes)
92 {
93 context.Scope.DefineVariable(type.Name, type);
94 }
95 }

97 private void SetupStage(string file, bool optimize = false, bool compile = false)
98 {
99 //Load expressions
100 IEnumerable<BadExpression> expressions = BadSourceParser.Create(file, BadFileSystem.ReadAllText(file)).Parse();
101
102 if (optimize)
103 {
104 expressions = BadExpressionOptimizer.Optimize(expressions);
105 }
106
107 //Create Context
108 BadExecutionContext context = BadExecutionContext.Create();
109
110 //Add Apis
111 LoadApis(context);
112
113 context.Run(expressions);
114 }
