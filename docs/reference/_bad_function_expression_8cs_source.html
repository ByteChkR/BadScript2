172 }

175 BadExpressionFunction f = new BadExpressionFunction(context.Scope,
176 Name,
177 m_Body,
178 m_Parameters.Select(x => x.Initialize(context)).ToArray(),
179 Position,
180 IsConstantFunction,
181 m_MetaData);
182
183 BadFunction fFinal = f;
184
185 if (CompileLevel == BadFunctionCompileLevel.Compiled)
186 {
187 fFinal = BadCompilerApi.CompileFunction(BadCompiler.Instance, f, true);
188 }
189 else if (CompileLevel == BadFunctionCompileLevel.CompiledFast)
190 {
191 fFinal = BadCompilerApi.CompileFunction(BadCompiler.Instance, f, false);
192 }
193
194
195 if (Name != null)
196 {
197 context.Scope.DefineVariable(BadObject.Wrap(Name.Text), fFinal);
198 }
