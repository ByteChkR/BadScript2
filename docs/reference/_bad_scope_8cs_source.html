222 private static string GetStackTrace(BadScope scope)
223 {
224 BadScope? current = scope;
225 List<BadScope> stack = new List<BadScope>();
226 while (current != null)
227 {
228 if (current.CountInStackTrace)
229 {
230 stack.Add(current);
231 }
232
233 current = current.Caller;
234 }
235
236 return string.Join("\n", stack.Select(s => s.Name));
