64 private static BadReflectedMemberTable CreateInternal(Type t)
65 {
66 Dictionary<string, BadReflectedMember> members = new Dictionary<string, BadReflectedMember>();
67
68 if (t.IsArray)
69 {
70 members.Add(BadStaticKeys.ArrayAccessOperatorName, new BadReflectedMethod(t.GetMethod("Get")));
71 }
72
73 foreach (MemberInfo info in t.GetMembers())
74 {
75 if (info is FieldInfo field)
76 {
77 members.Add(field.Name, new BadReflectedField(field));
78 }
79 else if (info is PropertyInfo property)
80 {
81 if (property.Name == "Item" && property.GetIndexParameters().Length > 0)
82 {
83 if (!members.ContainsKey(BadStaticKeys.ArrayAccessOperatorName))
84 {
85 members.Add(BadStaticKeys.ArrayAccessOperatorName, new BadReflectedMethod(property.GetMethod));
86 }
87 }
88 else
89 {
90 members.Add(property.Name, new BadReflectedProperty(property));
91 }
92 }
93 else if (info is MethodInfo method)
94 {
95 if (method.Name == "GetEnumerator")
96 {
97 members["GetEnumerator"] = new BadReflectedEnumeratorMethod(method);
98 }
99 else if (members.ContainsKey(method.Name) && members[method.Name] is BadReflectedMethod m)
100 {
101 m.AddMethod(method);
102 }
103 else
104 {
105 members.Add(method.Name, new BadReflectedMethod(method));
106 }
107 }
108 }
109
110 return new BadReflectedMemberTable(members);
