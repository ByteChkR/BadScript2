116 private void Write()
117 {

120 if (!m_Client!.Connected)
121 {
122 break;
123 }
124
125 if (m_Client.Available != 0)
126 {
127 byte[] len = new byte[sizeof(int)];
128 NetworkStream stream = m_Client.GetStream();
129 int read = stream.Read(len, 0, len.Length);
130
131 if (read != len.Length)

133 throw new BadNetworkConsoleException("Invalid Packet Size");

177 while (!m_ExitRequested)
178 {

180
181 while (!m_ExitRequested && !task.IsCompleted)
182 {
183 if (lastHeartBeat + TimeSpan.FromMilliseconds(HeartBeatSendInterval) < DateTime.Now)

188
189 Thread.Sleep(ConsoleReadInputSleep);

191
192 if (m_ExitRequested)

194 break;
195 }
196

198
199

202 int idx = message.IndexOf(' ');
203
204 if (idx == -1)
205 {
206 idx = message.Length;
207 }
208
209 string cmdName = message.Substring(0, idx).Remove(0, "client::".Length);
210 string args = message.Remove(0, idx).Trim();
211 BadNetworkConsoleClientCommand? cmd = m_Commands.FirstOrDefault(x => x.Name == cmdName);
212
213 if (cmd == null)
214 {
215 BadConsole.WriteLine("Command not found: " + cmdName);
216 }
217 else
218 {
219 cmd.Invoke(args);
220 }
221

223 }

225 lastHeartBeat = DateTime.Now;
226 byte[] packetData = new BadConsoleReadPacket(message).Serialize();

231 }
