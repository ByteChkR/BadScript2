305 [Test]
306 [TestCaseSource(nameof(GetHtmlTemplateFiles))]
307 public void TestHtmlTemplateDefault(string file)
308 {
309 BadHtmlTemplateOptions options = new BadHtmlTemplateOptions();
310 string referenceFile = Path.ChangeExtension(file, $".{(options.SkipEmptyTextNodes ? "skip" : "default")}.html");
311 HtmlDocument result = BadHtmlTemplate.Create(file).RunTemplate(null, options);
312
313 if (BadFileSystem.Instance.Exists(referenceFile))
314 {
315 HtmlDocument reference = LoadReference(referenceFile);
316 Assert.That(result.DocumentNode.OuterHtml, Is.EqualTo(reference.DocumentNode.OuterHtml));
317 }
318 else
319 {
320 BadFileSystem.Instance.CreateDirectory(Path.GetDirectoryName(referenceFile)!, true);
321 result.Save(referenceFile);
322 Assert.Inconclusive("Reference file does not exist. Saving...");
323 }
324 }
