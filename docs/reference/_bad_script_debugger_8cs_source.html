41 public void Step(BadDebuggerStep stepInfo)
42 {
43 stepInfo.GetSourceView(out int _, out int lineInSource);
44
45 if (m_LineNumbers.ContainsKey(stepInfo.Position.Source) &&
46 lineInSource == m_LineNumbers[stepInfo.Position.Source])
47 {
48 return;
49 }
50
51 m_LineNumbers[stepInfo.Position.Source] = lineInSource;
52
53 if (!m_SeenFiles.Contains(stepInfo.Position.FileName ?? ""))
54 {
55 m_SeenFiles.Add(stepInfo.Position.FileName ?? "");
56 OnFileLoaded?.Invoke(stepInfo.Position.FileName ?? "");
57 }
58
59 OnStep?.Invoke(stepInfo);
60 }
