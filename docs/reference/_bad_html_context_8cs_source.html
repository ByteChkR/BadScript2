153 public BadExpression[] Parse(string code, BadSourcePosition pos)
154 {
155 try
156 {
157 BadExpression[] expressions = BadSourceParser.Parse(FilePath, code).ToArray();
158
159 foreach (BadExpression expression in VisitAll(expressions))
160 {
161 BadSourcePosition newPosition = BadSourcePosition.Create(FilePath,
162 Source,
163 pos.Index + expression.Position.Index,
164 expression.Position.Length);
165 expression.SetPosition(newPosition);
166 }
167
168 return expressions;
169 }
170 catch (BadSourceReaderException e)
171 {
172 if (e.Position == null)
173 {
174 throw new BadSourceReaderException(e.OriginalMessage, pos);
175 }
176
177 throw new BadSourceReaderException(e.OriginalMessage,
178 BadSourcePosition.Create(FilePath, Source, e.Position.Index + pos.Index, e.Position.Length));
179 }
180 }
