84 protected override int Run(BadRunSystemSettings settings)
85 {
86 BadExecutionContextOptions options = CreateOptions();
87
88 BadNetworkConsoleHost? host = null;
89 if (settings.RemotePort != -1)
90 {
91 host = new BadNetworkConsoleHost(new TcpListener(IPAddress.Any, settings.RemotePort));
92 host.Start();
93 BadConsole.SetConsole(host);
94 }
95
96 BadRuntimeApi.StartupArguments = settings.Args;
97 IEnumerable<string> files = BadFileSystem.Instance.GetFiles(
98 StartupDirectory,
99 $".{BadRuntimeSettings.Instance.FileExtension}",
100 true
101 )
102 .Concat(settings.Files);
103 if (settings.Interactive)
104 {
105 if (settings.Benchmark)
106 {
107 BadLogger.Warn("Benchmarking is not supported in interactive mode");
108 }
109
110 int r= RunInteractive(options, files);
111
112 host?.Stop();
113 return r;
114
115 }
116
117 Stopwatch? sw = null;
118 if (settings.Benchmark)
119 {
120 sw = Stopwatch.StartNew();
121 }
122
123 if (settings.Debug)
124 {
125 BadDebugger.Attach(new BadScriptDebugger(options));
126 }
127
128 foreach (string file in files)
129 {
130 BadSourceParser parser = BadSourceParser.Create(file, BadFileSystem.ReadAllText(file));
131 BadExecutionContext context = options.Build();
132
133 context.Scope.AddSingleton(BadTaskRunner.Instance);
134
135 IEnumerable<BadExpression> exprs = parser.Parse();
136 if (BadNativeOptimizationSettings.Instance.UseConstantExpressionOptimization)
137 {
138 exprs = BadExpressionOptimizer.Optimize(exprs);
139 }
140
141 BadTaskRunner.Instance.AddTask(
142 new BadTask(new BadInteropRunnable(Run(context, context.Execute(exprs)).GetEnumerator()), "Main"),
143 true
144 );
145
146
147 while (!BadTaskRunner.Instance.IsIdle)
148 {
149 BadTaskRunner.Instance.RunStep();
150 }
151 }
152
153 if (settings.Benchmark)
154 {
155 sw?.Stop();
156 BadLogger.Log($"Execution Time: {sw?.ElapsedMilliseconds ?? 0}ms", "Benchmark");
157 }
158
159 host?.Stop();
160
161 return -1;
162 }
