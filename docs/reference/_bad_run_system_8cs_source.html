103 protected override int Run(BadRunSystemSettings settings)
104 {
105 BadExecutionContextOptions options = CreateOptions();
106
107 BadNetworkConsoleHost? host = null;
108
109 if (settings.RemotePort != -1)

112 host.Start();

114 }
115
116 BadRuntimeApi.StartupArguments = settings.Args;
117 IEnumerable<string> files = BadFileSystem.Instance.GetFiles(StartupDirectory,
118 $".{BadRuntimeSettings.Instance.FileExtension}",
119 true)

121
122 if (settings.Interactive)
123 {
124 if (settings.Benchmark)

126 BadLogger.Warn("Benchmarking is not supported in interactive mode");
127 }
128

131 host?.Stop();
132

140 sw = Stopwatch.StartNew();
141 }
142
143 if (settings.Debug)
144 {
145 BadDebugger.Attach(new BadScriptDebugger(options));
146 }
147
148 foreach (string file in files)
149 {
150 BadSourceParser parser = BadSourceParser.Create(file, BadFileSystem.ReadAllText(file));
151 BadExecutionContext context = options.Build();
152
153 context.Scope.AddSingleton(BadTaskRunner.Instance);
154
155 IEnumerable<BadExpression> exprs = parser.Parse();
156
157 if (BadNativeOptimizationSettings.Instance.UseConstantExpressionOptimization)
158 {
159 exprs = BadExpressionOptimizer.Optimize(exprs);
160 }
161
162 BadTaskRunner.Instance.AddTask(
163 new BadTask(new BadInteropRunnable(Run(context, context.Execute(exprs)).GetEnumerator()), "Main"),
164 true);
165
166
167 while (!BadTaskRunner.Instance.IsIdle)
168 {
169 BadTaskRunner.Instance.RunStep();
170 }
171 }
172
173 if (settings.Benchmark)
174 {
175 sw?.Stop();
176 BadLogger.Log($"Execution Time: {sw?.ElapsedMilliseconds ?? 0}ms", "Benchmark");
177 }
178
179 host?.Stop();
180
181 return -1;
