81 private BadExpression? ParsePrefix(int precedence)
82 {
83 // Parse Symbol
84 BadSymbolToken? symbol = Reader.TryParseSymbols(m_Operators.UnaryPrefixSymbols);
85
86 if (symbol == null)
87 {
88 return null;
89 }
90
91 BadUnaryPrefixOperator? op = m_Operators.FindUnaryPrefixOperator(symbol.Text);
92
93
94 if (op == null || op.Precedence > precedence)
95 {
96 Reader.SetPosition(symbol.SourcePosition.Index);
97
98 return null;
99 }
100
101 return op.Parse(this);
102 }

108 private BadExpression ParseIf()
109 {
110 int start = Reader.CurrentIndex;
111
112 Dictionary<BadExpression, BadExpression[]> conditionMap = new Dictionary<BadExpression, BadExpression[]>();
113
114 bool readNext;
115 bool isElse;
116
117 do
118 {
119 readNext = false;
120 isElse = false;
121 Reader.Eat(BadStaticKeys.IfKey);
122 Reader.SkipNonToken();
123 Reader.Eat('(');
124 Reader.SkipNonToken();
125 BadExpression condition = ParseExpression();
126 Reader.SkipNonToken();
127 Reader.Eat(')');
128 Reader.SkipNonToken();
129 List<BadExpression> block = ParseBlock(start, out bool _);
130 Reader.SkipNonToken();
131 conditionMap[condition] = block.ToArray();
132
133 if (Reader.Is(BadStaticKeys.ElseKey))
134 {
135 Reader.Eat(BadStaticKeys.ElseKey);
136 isElse = true;
137 Reader.SkipNonToken();
138 readNext = Reader.Is(BadStaticKeys.IfKey);
139 }
140 }
141 while (readNext);
142
143 BadExpression[]? elseBlock = null;
144
145 if (isElse)
146 {
147 elseBlock = ParseBlock(start, out bool _).ToArray();
148 }
149
150 return new BadIfExpression(conditionMap,
151 elseBlock,
152 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start));
153 }

240 private void ParseMeta()
241 {
242 if (Reader.Is("@|"))
243 {
244 Reader.Eat("@|");
245 Reader.SkipNonToken();
246 StringBuilder rootMeta = new StringBuilder();
247 StringBuilder returnMeta = new StringBuilder();
248 string returnType = "any";
249 Dictionary<string, (string, StringBuilder)> meta = new Dictionary<string, (string, StringBuilder)>();
250
251 StringBuilder GetMeta(string name, string type)
252 {
253 if (meta.TryGetValue(name, out (string, StringBuilder) val))
254 {
255 return val.Item2;
256 }
257
258 val = (type, new StringBuilder());
259 meta[name] = val;
260
261 return val.Item2;
262 }
263
264 while (!Reader.Is("|@"))
265 {
266 if (Reader.Is("|PARAM"))
267 {
268 Reader.Eat("|PARAM");
269 Reader.SkipNonToken();
270 string name = Reader.ParseWord().Text;
271 Reader.SkipNonToken();
272 string type = "any";
273
274 if (!Reader.Is(':'))
275 {
276 type = "";
277
278 while (!Reader.IsEof() && (Reader.IsWordStart() || Reader.Is('.')))
279 {
280 type += Reader.ParseWord().Text;
281
282 if (Reader.Is('.'))
283 {
284 type += '.';
285 Reader.MoveNext();
286 }
287 }
288
289 Reader.SkipNonToken();
290 }
291
292 Reader.Eat(':');
293 Reader.SkipNonToken();
294 StringBuilder m = GetMeta(name, type);
295
296 while (!Reader.IsEof() && !Reader.Is("|@") && !Reader.IsNewLine())
297 {
298 m.Append(Reader.GetCurrentChar());
299 Reader.MoveNext();
300 }
301
302 Reader.SkipNonToken();
303 }
304 else if (Reader.Is("|RET"))
305 {
306 Reader.Eat("|RET");
307 Reader.SkipNonToken();
308 string type = "any";
309
310 if (!Reader.Is(':'))
311 {
312 type = "";
313
314 while (!Reader.IsEof() && (Reader.IsWordStart() || Reader.Is('.')))
315 {
316 type += Reader.ParseWord().Text;
317
318 if (Reader.Is('.'))
319 {
320 type += '.';
321 Reader.MoveNext();
322 }
323 }
324
325 Reader.SkipNonToken();
326 }
327
328 Reader.Eat(':');
329 Reader.SkipNonToken();
330
331 while (!Reader.IsEof() && !Reader.Is("|@") && !Reader.IsNewLine())
332 {
333 returnMeta.Append(Reader.GetCurrentChar());
334 Reader.MoveNext();
335 }
336
337 Reader.SkipNonToken();
338
339 returnType = type;
340 }
341 else
342 {
343 if (Reader.IsNewLine(-1))
344 {
345 Reader.SkipNonToken();
346
347 if (Reader.Is("|"))
348 {
349 continue;
350 }
351 }
352
353 while (!Reader.IsEof() && !Reader.IsNewLine() && !Reader.Is("|@"))
354 {
355 rootMeta.Append(Reader.GetCurrentChar());
356 Reader.MoveNext();
357 }
358
359 rootMeta.AppendLine();
360 Reader.SkipNonToken();
361 }
362 }
363
364 Reader.Eat("|@");
365 Reader.SkipNonToken();
366
367 m_MetaData = new BadMetaData(rootMeta.ToString().Trim(),
368 returnMeta.ToString().Trim(),
369 returnType,
370 meta.ToDictionary(x => x.Key,
371 x => new BadParameterMetaData(x.Value.Item1,
372 x.Value.Item2.ToString().Trim())));
373 }

383 private BadExpression ParseValue(int precedence)
384 {
385 Reader.SkipNonToken();
386
387 ParseMeta();
388
389 Reader.SkipNonToken();
390
391 BadExpression? prefixExpr = ParsePrefix(precedence);
392
393 if (prefixExpr != null)
394 {
395 return prefixExpr;
396 }
397
398 if (Reader.Is('('))
399 {
400 int start = Reader.CurrentIndex;
401
402 try
403 {
404 List<BadFunctionParameter> p = ParseParameters(start);
405
406 Reader.SkipNonToken();
407
408 if (!Reader.Is("=>"))
409 {
410 Reader.SetPosition(start);
411 }
412 else
413 {
414 return ParseFunction(start, null, null, false, BadFunctionCompileLevel.None, p);
415 }
416 }
417 catch (Exception)
418 {
419 Reader.SetPosition(start);
420 }
421 }
422
423 if (Reader.Is('('))
424 {
425 Reader.Eat('(');
426 BadExpression expr = ParseExpression();
427 Reader.Eat(')');
428
429 return expr;
430 }
431
432 if (Reader.Is('['))
433 {
434 int start = Reader.CurrentIndex;
435 Reader.Eat('[');
436 Reader.SkipNonToken();
437 List<BadExpression> initExpressions = new List<BadExpression>();
438
439 if (!Reader.Is(']'))
440 {
441 bool parseNext;
442
443 do
444 {
445 parseNext = false;
446
447 Reader.SkipNonToken();
448 initExpressions.Add(ParseExpression());
449
450 Reader.SkipNonToken();
451
452 if (Reader.Is(','))
453 {
454 Reader.Eat(',');
455 Reader.SkipNonToken();
456 parseNext = true;
457 }
458 }
459 while (parseNext);
460 }
461
462 Reader.SkipNonToken();
463 Reader.Eat(']');
464
465 return new BadArrayExpression(initExpressions.ToArray(),
466 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start));
467 }
468
469 if (Reader.Is('{'))
470 {
471 int start = Reader.CurrentIndex;
472 Reader.Eat('{');
473 Dictionary<BadWordToken, BadExpression> table = new Dictionary<BadWordToken, BadExpression>();
474 Reader.SkipNonToken();
475
476 if (!Reader.Is('}'))
477 {
478 bool parseNext;
479
480 do
481 {
482 parseNext = false;
483 Reader.SkipNonToken();
484 BadWordToken key = Reader.ParseWord();
485 Reader.SkipNonToken();
486 Reader.Eat(':');
487 Reader.SkipNonToken();
488 BadExpression value = ParseExpression();
489 table[key] = value;
490 Reader.SkipNonToken();
491
492 if (Reader.Is(','))
493 {
494 parseNext = true;
495 Reader.Eat(',');
496 }
497 }
498 while (parseNext);
499 }
500
501 Reader.Eat('}');
502
503 return new BadTableExpression(table,
504 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start));
505 }

507 if (Reader.Is(BadStaticKeys.LockKey))
508 {
509 return ParseLock();
510 }
511
512 if (Reader.Is(BadStaticKeys.ForEachKey))
513 {
514 return ParseForEach();
515 }
516

518 {
519 return ParseFor();
520 }
521
522 if (Reader.Is(BadStaticKeys.IfKey))

524 return ParseIf();
525 }
526

528 {
529 BadSourcePosition pos = Reader.Eat(BadStaticKeys.ContinueKey);
530
531 if (Reader.IsWordChar())
532 {
533 Reader.SetPosition(pos.Index);
534 }
535 else
536 {
537 Reader.SkipNonToken();

540 }
541 }

546
547 if (Reader.IsWordChar())
548 {
549 Reader.SetPosition(pos.Index);
550 }
551 else
552 {
553 Reader.SkipNonToken();
554
555 return new BadBreakExpression(pos);
556 }
557 }
558

560 {
561 BadSourcePosition pos = Reader.Eat(BadStaticKeys.ThrowKey);
562

564 {
565 Reader.SetPosition(pos.Index);
566 }
567 else
568 {
569 return new BadThrowExpression(ParseExpression(), pos);
570 }
571 }
572
573 if (Reader.Is(BadStaticKeys.ReturnKey))
574 {
575 BadSourcePosition pos = Reader.Eat(BadStaticKeys.ReturnKey);
576
577 if (Reader.IsWordChar())
578 {
579 Reader.SetPosition(pos.Index);
580 }
581 else
582 {
583 Reader.SkipNonToken();
584 bool isRef = false;
585
586 if (Reader.Is(';'))
587 {
588 return new BadReturnExpression(null, pos, false);
589 }
590
591 if (Reader.Is(BadStaticKeys.RefKey))
592 {
593 isRef = true;
594 Reader.Eat(BadStaticKeys.RefKey);
595 Reader.SkipNonToken();
596 }
597

599

601 }

608

613 else

622

626 Reader.Eat(BadStaticKeys.ConstantDefinitionKey);

628 }

657 else

666 }

668 if (Reader.Is(BadStaticKeys.NewKey) && Reader.IsWhiteSpace(BadStaticKeys.NewKey.Length))

677

685 BadNullToken token = Reader.ParseNull();

688 {

690 }

692 {
693 return new BadNullExpression(token.SourcePosition);

695 }
696

698 {

700
701 return new BadStringExpression(token.Value, token.SourcePosition);
702 }
703
704 if (Reader.Is(BadStaticKeys.FormatStringKey))

706 return ParseFormatString();
707 }
708
709
710 if (Reader.Is(BadStaticKeys.MultiLineFormatStringKey))
711 {
712 return ParseMultiLineFormatString();
713 }
714
715
716 if (Reader.Is(BadStaticKeys.MultiLineStringKey))
717 {
718 BadStringToken token = Reader.ParseMultiLineString();
719
720 return new BadStringExpression(token.Value, token.SourcePosition);
721 }
722
723 if (Reader.IsNumberStart())
724 {
725 BadNumberToken token = Reader.ParseNumber();
726
727 return new BadNumberExpression(decimal.Parse(token.Text, NumberFormatInfo.InvariantInfo),
728 token.SourcePosition);
729 }
730
731 BadValueParser? valueParser = m_Operators.GetValueParser(this);
732
733 if (valueParser != null)
734 {
735 return valueParser.ParseValue(this);
736 }
737
738 Reader.SkipNonToken();
739
740 int wordStart = Reader.CurrentIndex;
741 BadWordToken word = Reader.ParseWord();
742
743 if (word.Text == BadStaticKeys.VariableDefinitionKey ||
744 word.Text == BadStaticKeys.ConstantDefinitionKey)
745 {
746 bool isReadOnly = word.Text == BadStaticKeys.ConstantDefinitionKey;
747 Reader.SkipNonToken();
748 BadExpression nameExpr = ParseValue(0);
749 Reader.SkipNonToken();
750
751 while (Reader.Is("."))
752 {
753 Reader.Eat(".");
754 BadWordToken right = Reader.ParseWord();
755 nameExpr = new BadMemberAccessExpression(nameExpr,
756 right,
757 nameExpr.Position.Combine(right.SourcePosition));
758 Reader.SkipNonToken();
759 }
760

763 Reader.SkipNonToken();
764

766 {
767 type = nameExpr;
768 name = Reader.ParseWord();
769 }
770 else
771 {
772 if (nameExpr is not BadVariableExpression expr)
773 {
774 throw new BadRuntimeException("Expected variable name", nameExpr.Position);
775 }
776

778 }
779
780 Reader.SkipNonToken();
781
782 return new BadVariableDefinitionExpression(name.Text, word.SourcePosition, type, isReadOnly);
783 }
784
785 Reader.SkipNonToken();

790

792 {
793 Reader.MoveNext();
794 }
795
796 Reader.SkipNonToken();
797
798 if (!Reader.Is("=>"))

801 }

806

820 {

822 }

826 null,

847 public BadExpression ParseExpression(BadExpression? left = null, int precedence = int.MaxValue)
848 {
849 left ??= ParseValue(precedence);
850
851 while (!Reader.IsEof())
852 {
853 Reader.SkipNonToken();
854
855 if (Reader.Is(BadStaticKeys.StatementEndKey))
856 {
857 return left;
858 }
859
860
861 if (Reader.Is('[') || Reader.Is("?["))
862 {
863 bool isNullChecked = Reader.Is("?[");
864
865 if (isNullChecked)
866 {
867 Reader.Eat("?[");
868 }
869 else
870 {
871 Reader.Eat('[');
872 }
873
874 Reader.SkipNonToken();
875 bool isReverse = false;
876
877 if (Reader.Is('^'))
878 {
879 Reader.Eat('^');
880 isReverse = true;
881 }
882
883 List<BadExpression> indices = new List<BadExpression>();
884
885 if (!Reader.Is(']'))
886 {
887 bool readNext;
888
889 do
890 {
891 readNext = false;
892 Reader.SkipNonToken();
893 indices.Add(ParseExpression());
894
895 Reader.SkipNonToken();
896
897 if (Reader.Is(','))
898 {
899 readNext = true;
900 Reader.Eat(',');
901 }
902
903 Reader.SkipNonToken();
904 }
905 while (readNext);
906 }
907
908 BadSourcePosition end = Reader.Eat(']');
909 Reader.SkipNonToken();
910
911 if (isReverse)
912 {
913 left = new BadArrayAccessReverseExpression(left,
914 indices.ToArray(),
915 left.Position.Combine(end),
916 isNullChecked);
917 }
918 else
919 {
920 left = new BadArrayAccessExpression(left,
921 indices.ToArray(),
922 left.Position.Combine(end),
923 isNullChecked);
924 }
925
926 continue;
927 }
928
929 if (Reader.Is('('))
930 {
931 Reader.Eat('(');
932 Reader.SkipNonToken();
933 List<BadExpression> args = new List<BadExpression>();
934
935 if (!Reader.Is(')'))
936 {
937 bool readNext;
938
939 do
940 {
941 readNext = false;
942 Reader.SkipNonToken();
943 args.Add(ParseExpression());
944
945 Reader.SkipNonToken();
946
947 if (Reader.Is(','))
948 {
949 readNext = true;
950 Reader.Eat(',');
951 }
952
953 Reader.SkipNonToken();
954 }
955 while (readNext);
956 }
957
958 BadSourcePosition rightPos = Reader.MakeSourcePosition(1);
959 Reader.Eat(')');
960 left = new BadInvocationExpression(left, args.ToArray(), left.Position.Combine(rightPos));
961 Reader.SkipNonToken();
962
963 continue;
964 }
965
966 // Parse Symbol
967 BadSymbolToken? symbol = Reader.TryParseSymbols(m_Operators.BinarySymbols);
968
969 if (symbol == null)
970 {
971 return left;
972 }
973
974 BadBinaryOperator? op = m_Operators.FindBinaryOperator(symbol.Text);
975
976
977 if (op == null || op.Precedence > precedence)
978 {
979 Reader.SetPosition(symbol.SourcePosition.Index);
980
981 return left;
982 }
983
984 left = op.Parse(left, this);
985 }
986
987 return left;
988 }

990 private BadStringExpression ParseMultiLineFormatString()
991 {
992 if (!Reader.Is(BadStaticKeys.MultiLineFormatStringKey))
993 {
994 throw new BadSourceReaderException(
995 $"Expected string start character but got '{(Reader.IsEof() ? "EOF" : Reader.GetCurrentChar())}'",
996 Reader.MakeSourcePosition(1));
997 }
998
999 int start = Reader.CurrentIndex;
1000 Reader.Eat(BadStaticKeys.MultiLineFormatStringKey);
1001 StringBuilder sb = new StringBuilder();
1002 List<BadExpression> args = new List<BadExpression>();
1003
1004 while (!Reader.IsStringQuote())
1005 {
1006 if (Reader.IsEof())
1007 {
1008 throw new BadSourceReaderException("String not terminated",
1009 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start));
1010 }
1011
1012 if (Reader.Is('{'))
1013 {
1014 Reader.Eat('{');
1015 Reader.SkipNonToken();
1016
1017 if (!Reader.Is('{'))
1018 {
1019 BadExpression expr = ParseExpression();
1020 Reader.SkipNonToken();
1021 Reader.Eat('}');
1022 sb.Append($"{{{args.Count}}}");
1023 args.Add(expr);
1024
1025 continue;
1026 }
1027
1028 Reader.Eat('{');
1029 sb.Append("{{");
1030
1031 continue;
1032 }
1033
1034 if (Reader.Is('}'))
1035 {
1036 int idx = Reader.CurrentIndex;
1037 Reader.Eat('}');
1038
1039 if (Reader.Is('}'))
1040 {
1041 Reader.Eat('}');
1042 sb.Append("}}");
1043
1044 continue;
1045 }
1046
1047 throw new BadParserException("Expected '}'",
1048 Reader.MakeSourcePosition(idx, 1));
1049 }
1050
1051 sb.Append(Reader.GetCurrentChar());
1052
1053 Reader.MoveNext();
