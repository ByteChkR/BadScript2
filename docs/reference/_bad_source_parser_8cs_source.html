69 private BadExpression? ParsePrefix(int precedence)
70 {
71 // Parse Symbol
72 BadSymbolToken? symbol = Reader.TryParseSymbols(m_Operators.UnaryPrefixSymbols);
73
74 if (symbol == null)
75 {
76 return null;
77 }
78
79 BadUnaryPrefixOperator? op = m_Operators.FindUnaryPrefixOperator(symbol.Text);
80
81
82 if (op == null || op.Precedence > precedence)
83 {
84 Reader.SetPosition(symbol.SourcePosition.Index);
85
86 return null;
87 }
88
89 return op.Parse(this);
90 }

96 private BadExpression ParseIf()
97 {
98 int start = Reader.CurrentIndex;
99
100 Dictionary<BadExpression, BadExpression[]> conditionMap = new Dictionary<BadExpression, BadExpression[]>();
101
102 bool readNext;
103 bool isElse;
104 do
105 {
106 readNext = false;
107 isElse = false;
108 Reader.Eat(BadStaticKeys.IfKey);
109 Reader.SkipNonToken();
110 Reader.Eat('(');
111 Reader.SkipNonToken();
112 BadExpression condition = ParseExpression();
113 Reader.SkipNonToken();
114 Reader.Eat(')');
115 Reader.SkipNonToken();
116 List<BadExpression> block = ParseBlock(start, out bool _);
117 Reader.SkipNonToken();
118 conditionMap[condition] = block.ToArray();
119
120 if (Reader.Is(BadStaticKeys.ElseKey))
121 {
122 Reader.Eat(BadStaticKeys.ElseKey);
123 isElse = true;
124 Reader.SkipNonToken();
125 readNext = Reader.Is(BadStaticKeys.IfKey);
126 }
127 }
128 while (readNext);
129
130 BadExpression[]? elseBlock = null;
131 if (isElse)
132 {
133 elseBlock = ParseBlock(start, out bool _).ToArray();
134 }
135
136 return new BadIfExpression(
137 conditionMap,
138 elseBlock,
139 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start)
140 );
141 }

237 private BadExpression ParseValue(int precedence)
238 {
239 Reader.SkipNonToken();
240
241 BadExpression? prefixExpr = ParsePrefix(precedence);
242
243 if (prefixExpr != null)
244 {
245 return prefixExpr;
246 }
247
248 if (Reader.Is('('))
249 {
250 Reader.Eat('(');
251 BadExpression expr = ParseExpression();
252 Reader.Eat(')');
253
254 return expr;
255 }
256
257 if (Reader.Is('['))
258 {
259 int start = Reader.CurrentIndex;
260 Reader.Eat('[');
261 Reader.SkipNonToken();
262 List<BadExpression> initExpressions = new List<BadExpression>();
263 if (!Reader.Is(']'))
264 {
265 bool parseNext;
266 do
267 {
268 parseNext = false;
269
270 Reader.SkipNonToken();
271 initExpressions.Add(ParseExpression());
272
273 Reader.SkipNonToken();
274
275 if (Reader.Is(','))
276 {
277 Reader.Eat(',');
278 Reader.SkipNonToken();
279 parseNext = true;
280 }
281 }
282 while (parseNext);
283 }
284
285 Reader.SkipNonToken();
286 Reader.Eat(']');
287
288 return new BadArrayExpression(
289 initExpressions.ToArray(),
290 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start)
291 );
292 }
293
294 if (Reader.Is('{'))
295 {
296 int start = Reader.CurrentIndex;
297 Reader.Eat('{');
298 Dictionary<BadWordToken, BadExpression> table = new Dictionary<BadWordToken, BadExpression>();
299 Reader.SkipNonToken();
300 if (!Reader.Is('}'))
301 {
302 bool parseNext;
303 do
304 {
305 parseNext = false;
306 Reader.SkipNonToken();
307 BadWordToken key = Reader.ParseWord();
308 Reader.SkipNonToken();
309 Reader.Eat(':');
310 Reader.SkipNonToken();
311 BadExpression value = ParseExpression();
312 table[key] = value;
313 Reader.SkipNonToken();
314 if (Reader.Is(','))
315 {
316 parseNext = true;
317 Reader.Eat(',');
318 }
319 }
320 while (parseNext);
321 }
322
323 Reader.Eat('}');
324
325 return new BadTableExpression(
326 table,
327 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start)
328 );

503 BadNumberToken token = Reader.ParseNumber();
504

506 }

509 if (valueParser != null)

511 return valueParser.ParseValue(this);
512 }
513
514 Reader.SkipNonToken();
515
516 BadWordToken word = Reader.ParseWord();
517
518 if (word.Text == BadStaticKeys.VariableDefinitionKey ||
519 word.Text == BadStaticKeys.ConstantDefinitionKey)
520 {
521 bool isReadOnly = word.Text == BadStaticKeys.ConstantDefinitionKey;
522 Reader.SkipNonToken();
523 BadExpression nameExpr = ParseValue(0);
524 Reader.SkipNonToken();
525
526 while (Reader.Is("."))
527 {
528 Reader.Eat(".");
529 BadWordToken right = Reader.ParseWord();
530 nameExpr = new BadMemberAccessExpression(
531 nameExpr,
532 right,
533 nameExpr.Position.Combine(right.SourcePosition)
534 );
535 Reader.SkipNonToken();
536 }
537
538 BadWordToken name;
539 BadExpression? type = null;
540 Reader.SkipNonToken();
541 if (Reader.IsWordStart())
542 {
543 type = nameExpr;
544 name = Reader.ParseWord();
545 }
546 else
547 {
548 if (nameExpr is not BadVariableExpression expr)
549 {
550 throw new BadRuntimeException("Expected variable name", nameExpr.Position);
551 }
552
553 name = expr.Name;
554 }
555
556 Reader.SkipNonToken();
557
558 return new BadVariableDefinitionExpression(name.Text, word.SourcePosition, type, isReadOnly);
559 }
560
561 return new BadVariableExpression(word.Text, word.SourcePosition);

570 public BadExpression ParseExpression(BadExpression? left = null, int precedence = int.MaxValue)
571 {
572 left ??= ParseValue(precedence);
573
574 while (!Reader.IsEof())
575 {
576 Reader.SkipNonToken();
577 if (Reader.Is(BadStaticKeys.StatementEndKey))
578 {
579 return left;
580 }
581
582
583 if (Reader.Is('[') || Reader.Is("?["))
584 {
585 bool isNullChecked = Reader.Is("?[");
586 if (isNullChecked)
587 {
588 Reader.Eat("?[");
589 }
590 else
591 {
592 Reader.Eat('[');
593 }
594
595 Reader.SkipNonToken();
596 bool isReverse = false;
597 if (Reader.Is('^'))
598 {
599 Reader.Eat('^');
600 isReverse = true;
601 }
602
603 List<BadExpression> indices = new List<BadExpression>();
604 if (!Reader.Is(']'))
605 {
606 bool readNext;
607 do
608 {
609 readNext = false;
610 Reader.SkipNonToken();
611 indices.Add(ParseExpression());
612
613 Reader.SkipNonToken();
614 if (Reader.Is(','))
615 {
616 readNext = true;
617 Reader.Eat(',');
618 }
619
620 Reader.SkipNonToken();
621 }
622 while (readNext);
623 }
624
625 BadSourcePosition end = Reader.Eat(']');
626 Reader.SkipNonToken();
627 if (isReverse)
628 {
629 left = new BadArrayAccessReverseExpression(left, indices.ToArray(), left.Position.Combine(end), isNullChecked);
630 }
631 else
632 {
633 left = new BadArrayAccessExpression(left, indices.ToArray(), left.Position.Combine(end), isNullChecked);
634 }
635
636 continue;
637 }
638
639 if (Reader.Is('('))
640 {
641 Reader.Eat('(');
642 Reader.SkipNonToken();
643 List<BadExpression> args = new List<BadExpression>();
644 if (!Reader.Is(')'))
645 {
646 bool readNext;
647 do
648 {
649 readNext = false;
650 Reader.SkipNonToken();
651 args.Add(ParseExpression());
652
653 Reader.SkipNonToken();
654 if (Reader.Is(','))
655 {
656 readNext = true;
657 Reader.Eat(',');
658 }
659
660 Reader.SkipNonToken();
661 }
662 while (readNext);
663 }
664
665 BadSourcePosition rightPos = Reader.MakeSourcePosition(1);
666 Reader.Eat(')');
667 left = new BadInvocationExpression(left, args.ToArray(), left.Position.Combine(rightPos));
668 Reader.SkipNonToken();
669
670 continue;
671 }
672
673 // Parse Symbol
674 BadSymbolToken? symbol = Reader.TryParseSymbols(m_Operators.BinarySymbols);
675
676 if (symbol == null)
677 {
678 return left;
679 }
680
681 BadBinaryOperator? op = m_Operators.FindBinaryOperator(symbol.Text);
682
683
684 if (op == null || op.Precedence > precedence)
685 {
686 Reader.SetPosition(symbol.SourcePosition.Index);
687
688 return left;
689 }
690
691 left = op.Parse(left, this);
692 }
693
694 return left;
695 }

710 private BadStringExpression ParseFormatString()
711 {
712 if (!Reader.Is(BadStaticKeys.FormatStringKey))
713 {
714 throw new BadSourceReaderException(
715 $"Expected string start character but got '{(Reader.IsEof() ? "EOF" : Reader.GetCurrentChar())}'",
716 Reader.MakeSourcePosition(1)
717 );
718 }
719
720 int start = Reader.CurrentIndex;
721 Reader.Eat(BadStaticKeys.FormatStringKey);
722 bool isEscaped = false;
723 StringBuilder sb = new StringBuilder();
724 List<BadExpression> args = new List<BadExpression>();
725 while (!Reader.IsStringQuote())
726 {
727 if (Reader.IsNewLine() || Reader.IsEof())
728 {
729 throw new BadSourceReaderException(
730 "String not terminated",
731 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start)
732 );
733 }
734
735 if (Reader.Is('{'))
736 {
737 Reader.Eat('{');
738 Reader.SkipNonToken();
739 if (!Reader.Is('{'))
740 {
741 BadExpression expr = ParseExpression();
742 Reader.SkipNonToken();
743 Reader.Eat('}');
744 sb.Append($"{{{args.Count}}}");
745 args.Add(expr);
746
747 continue;
748 }
749
750 Reader.Eat('{');
751 sb.Append("{{");
752
753 continue;
754 }
755
756 if (Reader.Is('}'))
757 {
758 int idx = Reader.CurrentIndex;
759 Reader.Eat('}');
760 if (Reader.Is('}'))
761 {
762 Reader.Eat('}');
763 sb.Append("}}");
764
765 continue;
766 }
767
768 throw new BadParserException(
769 "Expected '}'",
770 Reader.MakeSourcePosition(idx, 1)
771 );
772 }
773
774 if (Reader.Is(BadStaticKeys.EscapeCharacter))
775 {
776 isEscaped = true;
777 Reader.MoveNext();
778 }
779
780 if (isEscaped)
781 {
782 isEscaped = false;
783 sb.Append(Regex.Unescape($"\\{Reader.GetCurrentChar()}"));
784 }
785 else
786 {
787 sb.Append(Reader.GetCurrentChar());
788 }
789
790 Reader.MoveNext();
791 }
792
793 Reader.Eat(BadStaticKeys.Quote);
794
795 if (args.Count == 0)
796 {
797 return new BadStringExpression(
798 sb.ToString(),
799 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start)
800 );
801 }
802
803 return new BadFormattedStringExpression(
804 args.ToArray(),
805 sb.ToString(),
806 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start)
807 );
808 }

842 private List<BadExpression> ParseBlock(int start, out bool isSingleLine)
843 {
844 Reader.SkipNonToken();
845
846 List<BadExpression> block = new List<BadExpression>();
847 if (Reader.Is("=>"))
848 {
849 Reader.Eat("=>");
850 block.Add(ParseExpression());
851 isSingleLine = true;
852 }
853 else if (Reader.Is('{'))
854 {
855 isSingleLine = false;
856 Reader.Eat('{');
857 Reader.SkipNonToken();
858 while (!Reader.Is(BadStaticKeys.BlockEndKey))
859 {
860 Reader.SkipNonToken();
861 block.Add(ParseExpression());
862 Reader.SkipNonToken();
863 if (!Reader.Last(BadStaticKeys.BlockEndKey) && Reader.Is(BadStaticKeys.StatementEndKey))
864 {
865 Reader.Eat(BadStaticKeys.StatementEndKey);
866 }
867
868 Reader.SkipNonToken();
869 }
870
871 Reader.Eat(BadStaticKeys.BlockEndKey);
872 }
873 else
874 {
875 throw new BadParserException(
876 "Expected Expression Body",
877 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start)
878 );
879 }
880
881 Reader.SkipNonToken();
882
883 return block;
884 }

1004 private BadFunctionExpression ParseFunction(bool isConstant)
1005 {
1006 int start = Reader.CurrentIndex;
1007 Reader.Eat(BadStaticKeys.FunctionKey);
1008 Reader.SkipNonToken();
1009
1010 string? functionName = null;
1011 BadExpression? functionReturn = null;
1012 if (!Reader.Is('('))
1013 {
1014 BadExpression functionNameExpr = ParseValue(0);
1015 Reader.SkipNonToken();
1016
1017 while (Reader.Is("."))
1018 {
1019 Reader.Eat(".");
1020 BadWordToken right = Reader.ParseWord();
1021 functionNameExpr = new BadMemberAccessExpression(
1022 functionNameExpr,
1023 right,
1024 functionNameExpr.Position.Combine(right.SourcePosition)
1025 );
1026 Reader.SkipNonToken();
1027 }
1028
1029 Reader.SkipNonToken();
1030 if (!Reader.Is('('))
1031 {
1032 functionReturn = functionNameExpr;
1033 functionName = Reader.ParseWord().Text;
1034 }
1035 else
1036 {
1037 if (functionNameExpr is not BadVariableExpression expr)
1038 {
1039 throw new BadParserException(
1040 "Expected Variable Expression",
1041 functionNameExpr.Position
1042 );
1043 }
1044
1045 functionName = expr.Name;
1046 }
1047 }
1048
1049 Reader.Eat('(');
1050 Reader.SkipNonToken();
1051 List<BadFunctionParameter> parameters = new List<BadFunctionParameter>();
1052 if (!Reader.Is(')'))
1053 {
1054 bool first = true;
1055 bool hadOptional = false;
1056 bool hadRest = false;
1057 while (Reader.Is(',') || first)
1058 {
1059 if (hadRest)
1060 {
1061 throw new BadParserException(
1062 "Rest parameter must be last parameter",
1063 Reader.MakeSourcePosition(start, Reader.CurrentIndex - start)
1064 );
1065 }
1066
1067 if (!first)
1068 {
1069 Reader.Eat(',');
1070 Reader.SkipNonToken();
