860 public BadExpression ParseExpression(BadExpression? left = null, int precedence = int.MaxValue)
861 {
862 left ??= ParseValue(precedence);
863
864 while (!Reader.IsEof())
865 {
866 Reader.SkipNonToken();
867
868 if (Reader.Is(BadStaticKeys.StatementEndKey))
869 {
870 return left;
871 }
872
873
874 if (Reader.Is('[') || Reader.Is("?["))
875 {
876 bool isNullChecked = Reader.Is("?[");
877
878 if (isNullChecked)
879 {
880 Reader.Eat("?[");
881 }
882 else
883 {
884 Reader.Eat('[');
885 }
886
887 Reader.SkipNonToken();
888 bool isReverse = false;
889
890 if (Reader.Is('^'))
891 {
892 Reader.Eat('^');
893 isReverse = true;
894 }
895
896 List<BadExpression> indices = new List<BadExpression>();
897
898 if (!Reader.Is(']'))
899 {
900 bool readNext;
901
902 do
903 {
904 readNext = false;
905 Reader.SkipNonToken();
906 indices.Add(ParseExpression());
907
908 Reader.SkipNonToken();
909
910 if (Reader.Is(','))
911 {
912 readNext = true;
913 Reader.Eat(',');
914 }
915
916 Reader.SkipNonToken();
917 }
918 while (readNext);
919 }
920
921 BadSourcePosition end = Reader.Eat(']');
922 Reader.SkipNonToken();
923
924 if (isReverse)
925 {
926 left = new BadArrayAccessReverseExpression(left,
927 indices.ToArray(),
928 left.Position.Combine(end),
929 isNullChecked);
930 }
931 else
932 {
933 left = new BadArrayAccessExpression(left,
934 indices.ToArray(),
935 left.Position.Combine(end),
936 isNullChecked);
937 }
938
939 continue;
940 }
941
942 if (Reader.Is('('))
943 {
944 Reader.Eat('(');
945 Reader.SkipNonToken();
946 List<BadExpression> args = new List<BadExpression>();
947
948 if (!Reader.Is(')'))
949 {
950 bool readNext;
951
952 do
953 {
954 readNext = false;
955 Reader.SkipNonToken();
956 args.Add(ParseExpression());
957
958 Reader.SkipNonToken();
959
960 if (Reader.Is(','))
961 {
962 readNext = true;
963 Reader.Eat(',');
964 }
965
966 Reader.SkipNonToken();
967 }
968 while (readNext);
969 }
970
971 BadSourcePosition rightPos = Reader.MakeSourcePosition(1);
972 Reader.Eat(')');
973 left = new BadInvocationExpression(left, args.ToArray(), left.Position.Combine(rightPos));
974 Reader.SkipNonToken();
975
976 continue;
977 }
978
979 // Parse Symbol
980 BadSymbolToken? symbol = Reader.TryParseSymbols(m_Operators.BinarySymbols);
981
982 if (symbol == null)
983 {
984 return left;
985 }
986
987 BadBinaryOperator? op = m_Operators.FindBinaryOperator(symbol.Text);
988
989
990 if (op == null || op.Precedence > precedence)
991 {
992 Reader.SetPosition(symbol.SourcePosition.Index);
993
994 return left;
995 }
996
997 left = op.Parse(left, this);
998 }
999
1000 return left;
1001 }
