221 AllowEval = allowEval;

224 public IEnumerable<BadInstruction> Compile(BadExpression expression)
225 {
226 Type t = expression.GetType();
227
228 if (!m_Compilers.TryGetValue(t, out IBadExpressionCompiler compiler))
229 {
230 if (AllowEval)
231 {
232 return new[]
233 {
234 new BadInstruction(BadOpCode.Eval, expression.Position, expression)
235 };
236 }
237
238 throw new BadCompilerException("No Compiler for Expression Type " + expression.GetType().Name);
239 }
240
241 return compiler.Compile(this, expression);
242 }
243
244 public IEnumerable<BadInstruction> Compile(IEnumerable<BadExpression> expressions, bool clearStack = true)
245 {
246 foreach (BadExpression expression in expressions)
247 {
248 BadSourcePosition? position = null;
249
250 foreach (BadInstruction instruction in Compile(expression))
251 {
252 position = instruction.Position;
253
254 yield return instruction;
255 }
256
257 if (clearStack && position != null)
258 {
259 yield return new BadInstruction(BadOpCode.ClearStack, position);
260 }
261 }
262 }
263
264 public static IEnumerable<BadInstruction> Compile(string src)
265 {
266 BadSourceParser parser = BadSourceParser.Create("<nofile>", src);
267 BadCompiler compiler = new BadCompiler();
268
269 return compiler.Compile(parser.Parse());
270 }
