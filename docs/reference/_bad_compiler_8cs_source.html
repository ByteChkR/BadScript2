234 public BadCompiler(bool allowEval = true)
235 {
236 AllowEval = allowEval;
237 }
238
239 public IEnumerable<BadInstruction> Compile(BadExpression expression)
240 {
241 Type t = expression.GetType();
242
243 if (!m_Compilers.TryGetValue(t, out IBadExpressionCompiler compiler))
244 {
245 if (AllowEval)
246 {
247 return new[]
248 {
249 new BadInstruction(BadOpCode.Eval, expression.Position, expression)
250 };
251 }
252
253 throw new BadCompilerException("No Compiler for Expression Type " + expression.GetType().Name);
254 }
255
256 return compiler.Compile(this, expression);
257 }
258
259 public IEnumerable<BadInstruction> Compile(IEnumerable<BadExpression> expressions, bool clearStack = true)
260 {
261 foreach (BadExpression expression in expressions)
262 {
263 BadSourcePosition? position = null;
264
265 foreach (BadInstruction instruction in Compile(expression))
266 {
267 position = instruction.Position;
268
269 yield return instruction;
270 }
271
272 if (clearStack && position != null)
273 {
274 yield return new BadInstruction(BadOpCode.ClearStack, position);
275 }
276 }
277 }
278
279 public static IEnumerable<BadInstruction> Compile(string src)
280 {
281 BadSourceParser parser = BadSourceParser.Create("<nofile>", src);
282
283 return Instance.Compile(parser.Parse());
284 }
