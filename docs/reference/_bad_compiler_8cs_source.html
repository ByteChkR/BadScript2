92 public static int CompileExpression(BadExpression expr, BadCompilerResult result)
93 {
94 IBadExpressionCompiler? compiler = s_Compilers.FirstOrDefault(x => x.CanCompile(expr));
95 if (compiler == null)
96 {
97 throw new BadRuntimeException($"Cannot compile expression '{expr.GetType().Name}'", expr.Position);
98 }
99
100 return compiler.Compile(expr, result);
101 }

103 public static void CompileFunction(
104 BadFunctionExpression func,
105 out int funcStart,
106 out int funcLen,
107 BadCompilerResult result)
108 {
109 if (func.Name != null)
110 {
111 funcStart = result.DefineLabel(func.Name.Text);
112 }
113 else
114 {
115 funcStart = result.DefineLabel(result.UniqueLabel("func_"));
116 }
117
118 foreach (BadExpression expression in func.Body)
119 {
120 CompileExpression(expression, result);
121 result.Emit(new BadInstruction(BadOpCode.ClearStack, BadSourcePosition.FromSource("", 0, 0)));
122 }
123
124
125 result.Emit(new BadInstruction(BadOpCode.Push, BadSourcePosition.FromSource("", 0, 0), BadObject.Null));
126
127 result.Emit(new BadInstruction(BadOpCode.Return, BadSourcePosition.FromSource("", 0, 0)));
128
129 funcLen = result.GetInstructionLength() - funcStart;
130 }

173 public static BadCompiledFunction Compile(IEnumerable<BadExpression> block, BadScope parentScope)
174 {
175 BadCompilerResult result = new BadCompilerResult(parentScope);
176 CompileFunctionBlock(block, out int funcStart, out int funcLen, result);
177 result.ProcessWorkItems();
178
179 BadLogger.Log($"Compiled {result.GetInstructions().Length} Instructions", "Runtime");
180
181 return new BadCompiledFunction(
182 null,
183 parentScope,
184 BadSourcePosition.FromSource("", 0, 0),
185 result,
186 funcStart,
187 funcLen
188 );
189 }

191 public static BadFunction Compile(BadExpressionFunction func)
192 {
193 BadLogger.Log($"Compiling {func}", "Runtime");
194 BadCompilerResult result = new BadCompilerResult(func.ParentScope);
195 CompileFunction(func, out int start, out int len, result);
196 result.ProcessWorkItems();
197 BadCompiledFunction f = new BadCompiledFunction(
198 func.Name,
199 func.ParentScope,
200 func.Position,
201 result,
202 start,
203 len,
204 func.Parameters
205 );
206
207
208 BadLogger.Log($"Compiled {result.GetInstructions().Length} Instructions", "Runtime");
209
210 return f;
211 }
