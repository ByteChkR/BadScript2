62 private BadExpression? GetParameterType(BadHtmlContext context, HtmlAttribute attribute)
63 {
64 string name = attribute.Value;
65
66 while (name.EndsWith("*") || name.EndsWith("?") || name.EndsWith("!"))
67 {
68 name = name.Remove(name.Length - 1);
69 }
70

82 context.CreateAttributePosition(attribute));

84
85 return expressions[0];

88 public override void TransformNode(BadHtmlContext context)
89 {
90 HtmlAttribute? nameAttribute = context.InputNode.Attributes["name"];
91
92 if (nameAttribute == null)
93 {
94 throw BadRuntimeException.Create(context.ExecutionContext.Scope,
95 "Missing 'name' attribute in 'bs:function' node",
96 context.CreateOuterPosition());
97 }
98
99 if (string.IsNullOrEmpty(nameAttribute.Value))
100 {
101 throw BadRuntimeException.Create(context.ExecutionContext.Scope,
102 "Empty 'name' attribute in 'bs:function' node",
103 context.CreateAttributePosition(nameAttribute));
104 }
105
106 IEnumerable<HtmlAttribute> parameterAttributes =
107 context.InputNode.Attributes.Where(x => x.Name.StartsWith("param:"));
108
109 BadFunctionParameter[] parameters = parameterAttributes.Select(x => new BadFunctionParameter(
110 x.Name.Remove(0, "param:".Length),
111 IsOptional(x.Value),
112 IsNullChecked(x.Value),
113 IsRestArgs(x.Value),
114 GetParameterType(context, x)))
115 .ToArray();
116
117 BadInteropFunction func = new BadInteropFunction(nameAttribute.Value,
118 (ctx, args) => InvokeFunction(nameAttribute.Value, context, parameters, ctx, args),
119 parameters);
120 context.ExecutionContext.Scope.DefineVariable(nameAttribute.Value,
121 func,
122 context.ExecutionContext.Scope,
123 new BadPropertyInfo(func.GetPrototype(), true));
124 }

147 HtmlDocument outputDocument = new HtmlDocument();

153
