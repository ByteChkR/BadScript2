37 private BadExpression? GetParameterType(BadHtmlContext context, HtmlAttribute attribute)
38 {
39 string name = attribute.Value;
40
41 while (name.EndsWith("*") || name.EndsWith("?") || name.EndsWith("!"))
42 {
43 name = name.Remove(name.Length - 1);
44 }
45
46 if (string.IsNullOrEmpty(name))
47 {
48 return null;
49 }
50
51 BadExpression[] expressions = context.Parse(name, context.CreateAttributePosition(attribute));
52
53 if (expressions.Length != 1)
54 {
55 throw BadRuntimeException.Create(context.ExecutionContext.Scope,
56 $"Invalid parameter type expression for parameter {attribute.Name} in 'bs:function' node",
57 context.CreateAttributePosition(attribute));
58 }
59
60 return expressions[0];

63 public override void TransformNode(BadHtmlContext context)
64 {
65 HtmlAttribute? nameAttribute = context.InputNode.Attributes["name"];
66
67 if (nameAttribute == null)
68 {
69 throw BadRuntimeException.Create(context.ExecutionContext.Scope,
70 "Missing 'name' attribute in 'bs:function' node",
71 context.CreateOuterPosition());
72 }
73
74 if (string.IsNullOrEmpty(nameAttribute.Value))
75 {
76 throw BadRuntimeException.Create(context.ExecutionContext.Scope,
77 "Empty 'name' attribute in 'bs:function' node",
78 context.CreateAttributePosition(nameAttribute));
79 }
80
81 IEnumerable<HtmlAttribute> parameterAttributes =
82 context.InputNode.Attributes.Where(x => x.Name.StartsWith("param:"));
83
84 BadFunctionParameter[] parameters = parameterAttributes.Select(x => new BadFunctionParameter(
85 x.Name.Remove(0, "param:".Length),
86 IsOptional(x.Value),
87 IsNullChecked(x.Value),
88 IsRestArgs(x.Value),
89 GetParameterType(context, x)))
90 .ToArray();
91
92 BadInteropFunction func = new BadInteropFunction(nameAttribute.Value,
93 (ctx, args) => InvokeFunction(nameAttribute.Value, context, parameters, ctx, args),
94 parameters);
95 context.ExecutionContext.Scope.DefineVariable(nameAttribute.Value,
96 func,
97 context.ExecutionContext.Scope,
98 new BadPropertyInfo(func.GetPrototype(), true));
99 }

114 BadFunction.ApplyParameters(BadFunction.GetHeader(name, parameters),

129 }
