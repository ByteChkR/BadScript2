217 private static string ResolveEnvironmentVariables(BadSettings root, BadSettings parent, string str)
218 {
219 for (int i = 0; i < str.Length; i++)
220 {
221 if (str[i] == '$' && i + 1 < str.Length && str[i + 1] == '(')
222 {
223 int end = str.IndexOf(')', i + 2);
224 if (end == -1)
225 {
226 throw new Exception("Unclosed environment variable");
227 }
228
229 string envVar = str.Substring(i + 2, end - i - 2);
230
231 string? env = root.FindProperty<string>(envVar);
232 if (env == null)
233 {
234 env = parent.FindProperty<string>(envVar);
235 if (env == null)
236 {
237 throw new Exception($"Environment variable '{envVar}' not found");
238 }
239 }
240
241 str = str.Replace(str.Substring(i, end - i + 1), env);
242 i--;
243 }
244 }
245
246 return str;
