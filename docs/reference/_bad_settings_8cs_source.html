324 private static string ResolveEnvironmentVariables(BadSettings root, BadSettings parent, string str)
325 {
326 for (int i = 0; i < str.Length; i++)
327 {
328 if (str[i] == '$' && i + 1 < str.Length && str[i + 1] == '(')
329 {
330 int end = str.IndexOf(')', i + 2);
331 if (end == -1)
332 {
333 throw new Exception("Unclosed environment variable");
334 }
335
336 string envVar = str.Substring(i + 2, end - i - 2);
337
338 string? env = root.FindProperty<string>(envVar);
339 if (env == null)
340 {
341 env = parent.FindProperty<string>(envVar);
342 if (env == null)
343 {
344 throw new Exception($"Environment variable '{envVar}' not found");
345 }
346 }
347
348 str = str.Replace(str.Substring(i, end - i + 1), env);
349 i--;
350 }
351 }
