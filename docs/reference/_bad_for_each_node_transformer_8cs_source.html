45 {
46 HtmlAttribute? enumerationAttribute = context.InputNode.Attributes["on"];
47 HtmlAttribute? itemAttribute = context.InputNode.Attributes["as"];
48
49 if (enumerationAttribute == null)

53 context.CreateOuterPosition());
54 }
55
56 if (itemAttribute == null)
57 {
58 throw BadRuntimeException.Create(context.ExecutionContext.Scope,
59 "Missing 'as' attribute in 'bs:each' node",
60 context.CreateOuterPosition());
61 }
62
63 if (string.IsNullOrEmpty(enumerationAttribute.Value))
64 {
65 throw BadRuntimeException.Create(context.ExecutionContext.Scope,
66 "Empty 'on' attribute in 'bs:each' node",
67 context.CreateAttributePosition(enumerationAttribute));
68 }
69
70 if (string.IsNullOrEmpty(itemAttribute.Value))
71 {
72 throw BadRuntimeException.Create(context.ExecutionContext.Scope,
73 "Empty 'as' attribute in 'bs:each' node",
74 context.CreateAttributePosition(itemAttribute));
75 }
76
77 BadObject enumeration = context.ParseAndExecute(enumerationAttribute.Value,
78 context.CreateAttributePosition(enumerationAttribute));
79
80 if (enumeration is IBadEnumerable badEnumerable)
81 {
82 foreach (BadObject o in badEnumerable)
83 {
84 RunIteration(context, itemAttribute.Value, o);
85 }
86 }
87 else if (enumeration is IBadEnumerator badEnumerator)
88 {
89 while (badEnumerator.MoveNext())
90 {
91 RunIteration(context,
92 itemAttribute.Value,
93 badEnumerator.Current ??
94 throw BadRuntimeException.Create(context.ExecutionContext.Scope,
95 "Enumerator returned null",
96 context.CreateAttributePosition(enumerationAttribute)));
97 }
98 }
99 else if (enumeration is IEnumerable enumerable)
100 {
101 foreach (object? o in enumerable)
102 {
103 RunIteration(context, itemAttribute.Value, BadObject.Wrap(o));
104 }
105 }
106 else if (enumeration is IEnumerator enumerator)
107 {
108 while (enumerator.MoveNext())
109 {
110 RunIteration(context,
111 itemAttribute.Value,
112 BadObject.Wrap(enumerator.Current));
113 }
114 }
115 else
