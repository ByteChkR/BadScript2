25 private static BadObject Deflate(IBadString obj)
26 {
27 MemoryStream ms = new MemoryStream(Encoding.UTF8.GetBytes(obj.Value));
28 MemoryStream compressed = new MemoryStream();
29 using DeflateStream ds = new DeflateStream(compressed, CompressionMode.Compress);
30 ms.CopyTo(ds);
31 ds.Close();
32
33 return new BadArray(compressed.ToArray().Select(x => (BadObject)new BadNumber(x)).ToList());
34 }

126 private void ToDirectory(string outputDir, Stream input)
127 {
128 using ZipArchive archive = new ZipArchive(input, ZipArchiveMode.Read);
129 foreach (ZipArchiveEntry entry in archive.Entries)
130 {
131 string outputFile = Path.Combine(outputDir, entry.FullName);
132 string outputParent = Path.GetDirectoryName(outputFile)!;
133 m_FileSystem.CreateDirectory(outputParent, true);
134 using Stream s = m_FileSystem.OpenWrite(outputFile, BadWriteMode.CreateNew);
135 using Stream e = entry.Open();
136 e.CopyTo(s);
137 }
138 }
139
140 private void FromDirectory(string inputDir, Stream output)
141 {
142 using ZipArchive archive = new ZipArchive(output, ZipArchiveMode.Create);
143 string[] files = m_FileSystem.GetFiles(inputDir, "", true).ToArray();
144 foreach (string file in files)
145 {
146 string zipPath = file.Remove(0, inputDir.Length+1).Replace('\\', '/');
147 ZipArchiveEntry entry = archive.CreateEntry(zipPath);
148 using Stream es = entry.Open();
149 using Stream s = m_FileSystem.OpenRead(file);
150 s.CopyTo(es);
151 }
152 }
