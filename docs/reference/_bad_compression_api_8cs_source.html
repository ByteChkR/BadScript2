183 private void ToDirectory(string outputDir, Stream input)
184 {
185 using ZipArchive archive = new ZipArchive(input, ZipArchiveMode.Read);
186
187 foreach (ZipArchiveEntry entry in archive.Entries)
188 {
189 string outputFile = Path.Combine(outputDir, entry.FullName);
190 string outputParent = Path.GetDirectoryName(outputFile)!;
191 m_FileSystem.CreateDirectory(outputParent, true);
192 using Stream s = m_FileSystem.OpenWrite(outputFile, BadWriteMode.CreateNew);
193 using Stream e = entry.Open();
194 e.CopyTo(s);
195 }
196 }

203 private void FromDirectory(string inputDir, Stream output)
204 {
205 using ZipArchive archive = new ZipArchive(output, ZipArchiveMode.Create);
206 string[] files = m_FileSystem.GetFiles(inputDir, "", true).ToArray();
207
208 foreach (string file in files)
209 {
210 string zipPath = file.Remove(0, inputDir.Length + 1).Replace('\\', '/');
211 ZipArchiveEntry entry = archive.CreateEntry(zipPath);
212 using Stream es = entry.Open();
213 using Stream s = m_FileSystem.OpenRead(file);
214 s.CopyTo(es);
215 }
