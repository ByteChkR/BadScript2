16 // Executing Scripts with unknown content is a security risk if you expose APIs that allow scripts to interact with the host system.
17 // A notably dangerous api is the FileSystem API which allows scripts to read and write files on the host system.
18
19 // There a two ways to mitigate this risk:
20 // 1. Dont add any dangerous APIs to the execution context.
21 // 2. Use the Virtual FileSystem Abstraction. (this will be demonstrated here)
22
23 // The Virtual FileSystem Abstraction allows you to provide a virtual file system to the script.
24 // This virtual file system will be used by the script to read and write files.
25
26 // This script will be used to print the contents of "MySecretFile.txt" to the console.
27 string source = "Console.WriteLine(IO.File.ReadAllText(\"./MySecretFile.txt\"));";
28
29 //Add Common extensions
30 BadCommonInterop.AddExtensions();
31
32 // Lets create a "dangerous" context
33 BadExecutionContextOptions unsafeOptions = new BadExecutionContextOptions();
34
35 //Add Common APIs
36 unsafeOptions.AddApis(BadCommonInterop.Apis);
37
38 unsafeOptions.AddApi(new BadIOApi());
39
40 //This context exposes the FileSystem API which allows scripts to read and write files to the host machine.
41 BadExecutionContext unsafeContext = unsafeOptions.Build();
42
43 //Execute the Script. It will print out the contents of "MySecretFile.txt" to the console.
44 unsafeContext.Run(BadSourceParser.Parse("<none>", source));
45
46 // Now lets create a virtual file system
47 IFileSystem fileSystem = new BadVirtualFileSystem();
48
49 //Write the secret file to the virtual file system(just so the script does not crash when trying to access it)
50 fileSystem.WriteAllText("./MySecretFile.txt", "YIKES YOU ARE IN A SANDBOX");
51
52 BadExecutionContextOptions safeOptions = new BadExecutionContextOptions();
53
54 //Add Common APIs
55 safeOptions.AddApis(BadCommonInterop.Apis);
56
57 safeOptions.AddApi(new BadIOApi(fileSystem));
58
59 //This Context Exposes the FileSystem API but uses a virtual file system that is held in memory.
60 //The Script that is executed can only read and write files that are in the virtual file system.
61 BadExecutionContext safeContext = safeOptions.Build();
62
63 //Execute the Script. It will print out the contents of "MySecretFile.txt" to the console.
64 safeContext.Run(BadSourceParser.Parse("<none>", source));
65
66 // To make the virtual filesystem persistent you can export it to a zip file
67 using (FileStream fs = File.Create("MyVirtualFileSystem.zip"))
68 {
69 fileSystem.ExportZip(fs);
70 }
71
72 // You can also import a zip file to a virtual filesystem(this usually happens before a script is executed)
73 using (FileStream fs = File.OpenRead("MyVirtualFileSystem.zip"))
74 {
75 fileSystem.ImportZip(fs);
76 }
