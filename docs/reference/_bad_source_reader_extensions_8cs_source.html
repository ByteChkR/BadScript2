191 public static BadStringToken ParseString(this BadSourceReader reader)
192 {
193 if (!reader.IsStringQuote())
194 {
195 throw new BadSourceReaderException(
196 $"Expected string start character but got '{(reader.IsEOF() ? "EOF" : reader.GetCurrentChar())}'",
197 reader.MakeSourcePosition(1)
198 );
199 }
200
201 int start = reader.CurrentIndex;
202 reader.MoveNext();
203 bool isEscaped = false;
204 StringBuilder sb = new StringBuilder("\"");
205 while (!reader.IsStringQuote())
206 {
207 if (reader.IsNewLine() || reader.IsEOF())
208 {
209 throw new BadSourceReaderException(
210 "String not terminated",
211 reader.MakeSourcePosition(start, reader.CurrentIndex - start)
212 );
213 }
214
215 if (reader.Is(BadStaticKeys.EscapeCharacter))
216 {
217 isEscaped = true;
218 reader.MoveNext();
219 }
220
221 if (isEscaped)
222 {
223 isEscaped = false;
224 sb.Append(Regex.Unescape($"\\{reader.GetCurrentChar()}"));
225 }
226 else
227 {
228 sb.Append(reader.GetCurrentChar());
229 }
230
231 reader.MoveNext();
232 }
233
234 reader.Eat(BadStaticKeys.Quote);
235
236 sb.Append("\"");
237
238 return new BadStringToken(sb.ToString(), reader.MakeSourcePosition(start, reader.CurrentIndex - start));
239 }
