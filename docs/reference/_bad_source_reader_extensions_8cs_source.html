170 {
171 if (!reader.IsWordStart())
172 {
173 throw new BadSourceReaderException(

198 {
199 throw new BadSourceReaderException(

201 reader.MakeSourcePosition(1)
202 );
203 }
204
205 int start = reader.CurrentIndex;
206 reader.MoveNext();
207 bool hasDecimal = false;
208 while (reader.IsDigit() || !hasDecimal && reader.Is(BadStaticKeys.DecimalSeparator))

211 {
212 hasDecimal = true;
213 }

215 reader.MoveNext();
216 }
217
218 if (reader.Last(BadStaticKeys.DecimalSeparator))
219 {
220 reader.SetPosition(reader.CurrentIndex - 1);
221 }
222
223 return new BadNumberToken(reader.MakeSourcePosition(start, reader.CurrentIndex - start));

305 public static BadStringToken ParseString(this BadSourceReader reader)
306 {
307 if (!reader.IsStringQuote())
308 {
309 throw new BadSourceReaderException(
310 $"Expected string start character but got '{(reader.IsEof() ? "EOF" : reader.GetCurrentChar())}'",
311 reader.MakeSourcePosition(1)
312 );
313 }
314
315 int start = reader.CurrentIndex;
316 reader.MoveNext();
317 bool isEscaped = false;
318 StringBuilder sb = new StringBuilder("\"");
319 while (!reader.IsStringQuote())
320 {
321 if (reader.IsNewLine() || reader.IsEof())
322 {
323 throw new BadSourceReaderException(
324 "String not terminated",
325 reader.MakeSourcePosition(start, reader.CurrentIndex - start)
326 );
327 }
328
329 if (reader.Is(BadStaticKeys.EscapeCharacter))
330 {
331 isEscaped = true;
332 reader.MoveNext();
333 }
334
335 if (isEscaped)
336 {
337 isEscaped = false;
338 sb.Append(Regex.Unescape($"\\{reader.GetCurrentChar()}"));
339 }
340 else
341 {
342 sb.Append(reader.GetCurrentChar());
343 }
344
345 reader.MoveNext();
346 }
347
348 reader.Eat(BadStaticKeys.Quote);
349
350 sb.Append("\"");
351
352 return new BadStringToken(sb.ToString(), reader.MakeSourcePosition(start, reader.CurrentIndex - start));
353 }
