316 public static BadStringToken ParseString(this BadSourceReader reader)
317 {
318 if (!reader.IsStringQuote())
319 {
320 throw new BadSourceReaderException(
321 $"Expected string start character but got '{(reader.IsEof() ? "EOF" : reader.GetCurrentChar())}'",
322 reader.MakeSourcePosition(1));
323 }
324
325 int start = reader.CurrentIndex;
326 reader.MoveNext();
327 bool isEscaped = false;
328 StringBuilder sb = new StringBuilder("\"");
329
330 while (!reader.IsStringQuote())
331 {
332 if (reader.IsNewLine() || reader.IsEof())
333 {
334 throw new BadSourceReaderException("String not terminated",
335 reader.MakeSourcePosition(start, reader.CurrentIndex - start));
336 }
337
338 if (reader.Is(BadStaticKeys.EscapeCharacter))
339 {
340 isEscaped = true;
341 reader.MoveNext();
342 }
343
344 if (isEscaped)
345 {
346 isEscaped = false;
347 sb.Append(Regex.Unescape($"\\{reader.GetCurrentChar()}"));
348 }
349 else
350 {
351 sb.Append(reader.GetCurrentChar());
352 }
353
354 reader.MoveNext();
355 }
356
357 reader.Eat(BadStaticKeys.Quote);
358
359 sb.Append("\"");
360
361 return new BadStringToken(sb.ToString(), reader.MakeSourcePosition(start, reader.CurrentIndex - start));
362 }
