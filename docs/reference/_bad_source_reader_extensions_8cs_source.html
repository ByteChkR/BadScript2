305 public static BadStringToken ParseString(this BadSourceReader reader)
306 {
307 if (!reader.IsStringQuote())
308 {
309 throw new BadSourceReaderException(
310 $"Expected string start character but got '{(reader.IsEof() ? "EOF" : reader.GetCurrentChar())}'",
311 reader.MakeSourcePosition(1));
312 }
313
314 int start = reader.CurrentIndex;
315 reader.MoveNext();
316 bool isEscaped = false;
317 StringBuilder sb = new StringBuilder("\"");
318
319 while (!reader.IsStringQuote())
320 {
321 if (reader.IsNewLine() || reader.IsEof())
322 {
323 throw new BadSourceReaderException("String not terminated",
324 reader.MakeSourcePosition(start, reader.CurrentIndex - start));
325 }
326
327 if (reader.Is(BadStaticKeys.EscapeCharacter))
328 {
329 isEscaped = true;
330 reader.MoveNext();
331 }
332
333 if (isEscaped)
334 {
335 isEscaped = false;
336 sb.Append(Regex.Unescape($"\\{reader.GetCurrentChar()}"));
337 }
338 else
339 {
340 sb.Append(reader.GetCurrentChar());
341 }
342
343 reader.MoveNext();
344 }
345
346 reader.Eat(BadStaticKeys.Quote);
347
348 sb.Append("\"");
349
350 return new BadStringToken(sb.ToString(), reader.MakeSourcePosition(start, reader.CurrentIndex - start));
351 }
