42 private static void CopyHeaderTable(NameValueCollection dst, BadTable src)
43 {
44 foreach (KeyValuePair<BadObject, BadObject> kvp in src.InnerTable)
45 {
46 if (kvp.Key is not IBadString k || kvp.Value is not IBadString v)
47 {
48 throw new BadRuntimeException("Header Table must only contain string keys and values");
49 }

51 dst[k.Value] = v.Value;

115 private BadTable CreateResponseTable(HttpListenerResponse resp)
116 {
117 BadTable table = new BadTable();
118 table.SetFunction("SetHeader", (string key, string value) => resp.Headers[key] = value);
119 table.SetFunction("SetHeaders", (BadTable headers) => CopyHeaderTable(resp.Headers, headers));
120 table.SetFunction("SetCookie", (Cookie cookie) => resp.Cookies.Add(cookie));
121 table.SetFunction(
122 "SetCookies",
123 (BadArray cookies) =>
124 {
125 foreach (BadObject cookie in cookies.InnerArray)
126 {
127 if (cookie is BadReflectedObject cro && cro.Instance is Cookie c)
128 {
129 resp.Cookies.Add(c);
130 }
131 else
132 {
133 throw new BadRuntimeException("Cookies must be Cookie objects");
134 }
135 }
136 }
137 );
138 table.SetFunction("SetStatusCode", (int code) => resp.StatusCode = code);
139 table.SetFunction("SetStatusDescription", (string desc) => resp.StatusDescription = desc);
140 table.SetFunction("SetContentType", (string type) => resp.ContentType = type);
141 table.SetFunction("SetContentEncoding", (string enc) => resp.ContentEncoding = Encoding.GetEncoding(enc));
142 table.SetFunction("SetContentLength", (long len) => resp.ContentLength64 = len);
143 table.SetFunction("SetKeepAlive", (bool keepAlive) => resp.KeepAlive = keepAlive);
144 table.SetFunction("Redirect", (string location) => resp.Redirect(location));
145 table.SetFunction("Abort", resp.Abort);
146 table.SetFunction("Close", resp.Close);
147 table.SetFunction(
148 "SetContent",
149 (string content) =>
150 {
151 byte[] data = (resp.ContentEncoding ?? Encoding.UTF8).GetBytes(content);
152 resp.OutputStream.Write(data, 0, data.Length);
153 }
154 );
155
156 return table;
