27 public static bool InnerWhere(string varName, BadExpression query, object? o)
28 {
29 BadExecutionContext ctx = PredicateContextOptions.Build();
30 if (o is BadObject bo)
31 {
32 ctx.Scope.DefineVariable(varName, bo);
33 }
34 else
35 {
36 if (BadObject.CanWrap(o))
37 {
38 ctx.Scope.DefineVariable(varName, BadObject.Wrap(o));
39 }
40 else
41 {
42 ctx.Scope.DefineVariable(varName, new BadReflectedObject(o!));
43 }
44 }
45
46 BadObject r = BadObject.Null;
47 foreach (BadObject o1 in query.Execute(ctx))
48 {
49 if (ctx.Scope.IsError)
50 {
51 throw new Exception($"Error in LINQ Where: {varName} => {query} : {ctx.Scope.Error!.ToSafeString()}");
52 }
53
54 r = o1;
55 }
56
57 r = r.Dereference();
58 if (r is not IBadBoolean b)
59 {
60 throw new Exception($"Error in LINQ Where: {varName} => {query} : {r.ToSafeString()} is not a boolean");
61 }
62
63 return b.Value;
64 }
