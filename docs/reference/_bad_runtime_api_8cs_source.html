184 private static BadObject CompileSource(BadObject str, BadObject fileObj)
185 {
186 if (str is not IBadString src)
187 {
188 throw new BadRuntimeException($"Evaluate: Argument 'src' is not a string {str}");
189 }
190
191 string file = "<eval>";
192 if (fileObj is IBadString fileStr)
193 {
194 file = fileStr.Value;
195 }
196 else if (fileObj != BadObject.Null)
197 {
198 throw new BadRuntimeException("Evaluate: Argument 'fileObj' is not a string");
199 }
200
201 bool optimize = BadNativeOptimizationSettings.Instance.UseConstantExpressionOptimization;
202
203 BadExecutionContext ctx = BadExecutionContextOptions.Default.Build();
204 IEnumerable<BadExpression> exprs = BadSourceParser.Create(file, src.Value).Parse();
205 if (optimize)
206 {
207 exprs = BadExpressionOptimizer.Optimize(exprs);
208 }
209
210 BadLogger.Log($"Compiling file {file}", "Runtime");
211
212 return BadCompiler.Compile(exprs, ctx.Scope);
213 }
