161 private BadObject Evaluate(BadObject str, BadObject fileObj)
162 {
163 if (str is not IBadString src)
164 {
165 throw new BadRuntimeException($"Evaluate: Argument 'src' is not a string {str}");
166 }
167
168 string file = "<eval>";
169 if (fileObj is IBadString fileStr)
170 {
171 file = fileStr.Value;
172 }
173 else if (fileObj != BadObject.Null)
174 {
175 throw new BadRuntimeException("Evaluate: Argument 'fileObj' is not a string");
176 }
177
178
179 bool optimize = BadNativeOptimizationSettings.Instance.UseConstantExpressionOptimization;
180
181 BadExecutionContext ctx = BadExecutionContextOptions.Default.Build();
182
183 IEnumerable<BadExpression> exprs = BadSourceParser.Create(file, src.Value).Parse();
184 if (optimize)
185 {
186 exprs = BadExpressionOptimizer.Optimize(exprs);
187 }
188
189 return ctx.Run(exprs) ?? BadObject.Null;
190 }
