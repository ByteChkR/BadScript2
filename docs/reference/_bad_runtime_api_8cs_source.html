308 private BadObject Evaluate(
309 BadExecutionContext caller,
310 BadObject str,
311 BadObject fileObj,
312 BadObject optimizeExpr,
313 BadObject scope)
314 {
315 if (str is not IBadString src)
316 {
317 throw new BadRuntimeException($"Evaluate: Argument 'src' is not a string {str}");
318 }
319
320 if (optimizeExpr is not IBadBoolean optimizeE)
321 {
322 throw new BadRuntimeException($"Evaluate: Argument 'optimize' is not a boolean {optimizeExpr}");
323 }
324
325 string file = "<eval>";
326
327 if (fileObj is IBadString fileStr)
328 {
329 file = fileStr.Value;
330 }
331 else if (fileObj != BadObject.Null)
332 {
333 throw new BadRuntimeException("Evaluate: Argument 'fileObj' is not a string");
334 }
335
336
337 bool optimize = BadNativeOptimizationSettings.Instance.UseConstantExpressionOptimization && optimizeE.Value;
338
339 BadExecutionContext ctx =
340 scope == BadObject.Null || scope is not BadScope sc ?
341 new BadExecutionContext(caller.Scope.GetRootScope()
342 .CreateChild("<EvaluateAsync>", caller.Scope, null)) :
343 new BadExecutionContext(sc);
344
345 IEnumerable<BadExpression> exprs = BadSourceParser.Create(file, src.Value).Parse();
346
347 if (optimize)
348 {
349 exprs = BadExpressionOptimizer.Optimize(exprs);
350 }
351
352 return ctx.Run(exprs) ?? BadObject.Null;
353 }

402

406 .CreateChild("<EvaluateAsync>", caller.Scope, null)) :

410

415
