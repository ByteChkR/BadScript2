185 private BadObject Evaluate(BadExecutionContext caller, BadObject str, BadObject fileObj, BadObject optimizeExpr, BadObject scope)
186 {
187 if (str is not IBadString src)
188 {
189 throw new BadRuntimeException($"Evaluate: Argument 'src' is not a string {str}");
190 }
191
192 if (optimizeExpr is not IBadBoolean optimizeE)
193 {
194 throw new BadRuntimeException($"Evaluate: Argument 'optimize' is not a boolean {optimizeExpr}");
195 }
196
197 string file = "<eval>";
198 if (fileObj is IBadString fileStr)
199 {
200 file = fileStr.Value;
201 }
202 else if (fileObj != BadObject.Null)
203 {
204 throw new BadRuntimeException("Evaluate: Argument 'fileObj' is not a string");
205 }
206
207
208 bool optimize = BadNativeOptimizationSettings.Instance.UseConstantExpressionOptimization && optimizeE.Value;
209
210 BadExecutionContext ctx =
211 scope == BadObject.Null || scope is not BadScope sc
212 ? new BadExecutionContext(caller.Scope.GetRootScope().CreateChild("<EvaluateAsync>", caller.Scope))
213 : new BadExecutionContext(sc);
214
215 IEnumerable<BadExpression> exprs = BadSourceParser.Create(file, src.Value).Parse();
216 if (optimize)
217 {
218 exprs = BadExpressionOptimizer.Optimize(exprs);
219 }
220
221 return ctx.Run(exprs) ?? BadObject.Null;
222 }
