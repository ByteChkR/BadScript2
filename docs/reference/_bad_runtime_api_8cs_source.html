329 private BadObject Evaluate(
330 BadExecutionContext caller,
331 BadObject str,
332 BadObject fileObj,
333 BadObject optimizeExpr,
334 BadObject scope)
335 {
336 if (str is not IBadString src)
337 {
338 throw new BadRuntimeException($"Evaluate: Argument 'src' is not a string {str}");
339 }
340
341 if (optimizeExpr is not IBadBoolean optimizeE)
342 {
343 throw new BadRuntimeException($"Evaluate: Argument 'optimize' is not a boolean {optimizeExpr}");
344 }
345
346 string file = "<eval>";
347
348 if (fileObj is IBadString fileStr)
349 {
350 file = fileStr.Value;
351 }
352 else if (fileObj != BadObject.Null)
353 {
354 throw new BadRuntimeException("Evaluate: Argument 'fileObj' is not a string");
355 }
356
357
358 bool optimize = BadNativeOptimizationSettings.Instance.UseConstantExpressionOptimization && optimizeE.Value;
359
360 BadExecutionContext ctx =
361 scope == BadObject.Null || scope is not BadScope sc ?
362 new BadExecutionContext(caller.Scope.GetRootScope()
363 .CreateChild("<EvaluateAsync>", caller.Scope, null)) :
364 new BadExecutionContext(sc);
365
366 IEnumerable<BadExpression> exprs = BadSourceParser.Create(file, src.Value).Parse();
367
368 if (optimize)
369 {
370 exprs = BadExpressionOptimizer.Optimize(exprs);
371 }
372
373 return ctx.Run(exprs) ?? BadObject.Null;
374 }

421

423
424 BadExecutionContext ctx =
425 scope == BadObject.Null || scope is not BadScope sc ?
426 new BadExecutionContext(caller.Scope.GetRootScope()
427 .CreateChild("<EvaluateAsync>", caller.Scope, null)) :
428 new BadExecutionContext(sc);
429
430 IEnumerable<BadExpression> exprs = BadSourceParser.Create(file, src.Value).Parse();
431
432 if (optimize)
433 {
434 exprs = BadExpressionOptimizer.Optimize(exprs);
435 }
436
437
438 return new BadTask(
439 new BadInteropRunnable(SafeExecute(ctx.Execute(exprs), ctx).GetEnumerator(), setLastAsReturnB.Value),
440 "EvaluateAsync");

443 private IEnumerable<BadObject> SafeExecute(IEnumerable<BadObject> script, BadExecutionContext ctx)
444 {
445 foreach (BadObject o in script)
446 {
447 if (ctx.Scope.IsError)

449 throw new BadRuntimeErrorException(ctx.Scope.Error);
450 }
451
452 yield return o;
