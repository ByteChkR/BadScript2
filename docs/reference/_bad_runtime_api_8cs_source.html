181 private BadObject Evaluate(BadObject str, BadObject fileObj, BadObject optimizeExpr, BadObject scope)
182 {
183 if (str is not IBadString src)
184 {
185 throw new BadRuntimeException($"Evaluate: Argument 'src' is not a string {str}");
186 }
187
188 if (optimizeExpr is not IBadBoolean optimizeE)
189 {
190 throw new BadRuntimeException($"Evaluate: Argument 'optimize' is not a boolean {optimizeExpr}");
191 }
192
193 string file = "<eval>";
194 if (fileObj is IBadString fileStr)
195 {
196 file = fileStr.Value;
197 }
198 else if (fileObj != BadObject.Null)
199 {
200 throw new BadRuntimeException("Evaluate: Argument 'fileObj' is not a string");
201 }
202
203
204 bool optimize = BadNativeOptimizationSettings.Instance.UseConstantExpressionOptimization && optimizeE.Value;
205
206 BadExecutionContext ctx =
207 scope == BadObject.Null || scope is not BadScope sc ? BadExecutionContextOptions.Default.Build() : new BadExecutionContext(sc);
208
209 IEnumerable<BadExpression> exprs = BadSourceParser.Create(file, src.Value).Parse();
210 if (optimize)
211 {
212 exprs = BadExpressionOptimizer.Optimize(exprs);
213 }
214
215 return ctx.Run(exprs) ?? BadObject.Null;
216 }
