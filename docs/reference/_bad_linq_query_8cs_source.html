45 if (cmd.HasArgument)

50 !m_Commands.Any(x => x.Names.Any(y => linqQuery.StartsWith(y))))
51 {
52 sb.Append(linqQuery[0]);
53 linqQuery = linqQuery.Substring(1);
54 }
55
56 string arg = sb.ToString().Trim();
57
58 if (string.IsNullOrWhiteSpace(arg) && !cmd.IsArgumentOptional)
59 {

61 }
62
63 current = cmd.Run(new BadLinqQueryCommandData(current, arg));
64 }
