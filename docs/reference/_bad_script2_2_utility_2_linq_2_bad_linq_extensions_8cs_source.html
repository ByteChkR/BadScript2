28 private static object? InnerSelect(string varName, BadExpression query, object? o)
29 {
30 BadExecutionContext ctx = BadLinqCommon.PredicateContextOptions.Build();
31
32 if (o is BadObject bo)
33 {
34 ctx.Scope.DefineVariable(varName, bo);
35 }
36 else
37 {
38 if (BadObject.CanWrap(o))
39 {
40 ctx.Scope.DefineVariable(varName, BadObject.Wrap(o));
41 }
42 else
43 {
44 ctx.Scope.DefineVariable(varName, new BadReflectedObject(o!));
45 }
46 }
47
48 BadObject r = BadObject.Null;
49
50 foreach (BadObject o1 in query.Execute(ctx))
51 {
52 if (ctx.Scope.IsError)
53 {
54 throw new Exception($"Error in LINQ Select: {varName} => {query} : {ctx.Scope.Error!.ToSafeString()}");
55 }
56
57 r = o1;
58 }
59
60 return r.Dereference().Unpack();
61 }
