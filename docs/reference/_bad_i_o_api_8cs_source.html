135 t.SetFunction<string, BadArray>("WriteAllBytes",
136 (s, a) =>
137 {
138 using Stream stream = m_FileSystem.OpenWrite(s, BadWriteMode.CreateNew);
139
140 foreach (BadObject o in a.InnerArray)
141 {
142 if (o is not IBadNumber num)
143 {
144 throw new BadRuntimeException("BadIO.WriteAllBytes: Array contains non-number");
145 }
146
147 stream.WriteByte((byte)num.Value);
148 }
149 });
150
151 t.SetFunction<string>("ReadAllBytes",
152 s =>
153 {
154 using Stream stream = m_FileSystem.OpenRead(s);
155 byte[] bytes = new byte[stream.Length];
156 stream.Read(bytes, 0, bytes.Length);
157
158 return new BadArray(bytes.Select(x => (BadObject)new BadNumber(x)).ToList());
159 });
