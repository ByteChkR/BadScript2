32 private static async Task GenerateFoodLibrary()
33 {
34 //Load the template
35 BadHtmlTemplate template = BadHtmlTemplate.Create("templates/food.bhtml");
36
37 string url = "https://www.fruityvice.com/api/fruit/all";
38 using HttpClient client = new HttpClient();
39
40 BadObject jsonData = BadJson.FromJson(await client.GetStringAsync(url));
41
42 // All templates that depend on external data need a model object that the template can pull data from.
43 // The model object can be any object that BadScript2 can convert to a BadObject.
44 // The model object can also be a BadObject directly.
45 BadTable model = new BadTable();
46
47 //Set the Data
48 model.SetProperty("Fruits", jsonData);
49
50 //(Optional) Property that will be used to sort the fruit list.
51 model.SetProperty("Order", "id"); //Order By ID
52
53
54 //Run the template with the model and write the result to a file
55 File.WriteAllText("foodById.html", template.Run(model));
56
57
58 //(Optional) Property that will be used to sort the fruit list.
59 model.SetProperty("Order", "name"); //Order By Name
60
61 File.WriteAllText("foodByName.html", template.Run(model));
62 }
